\begin{frame}{Energiformer}\vfill
	\begin{block}{Kinetisk energi}
		Vi kender den kinetiske energi fra vores dagligdag. Det er denne energiform som knytter sig til bevægelse, hvorfor den på almindelig dansk oftest kaldes for ``Bevægelsesenergi''.  Energien opstår fordi et objekt har en hastighed.
	\end{block}\vfill
	\begin{block}{Potentiel energi}
		Som med den kinetiske energi, er vi bekendte med den potentielle energi fra vores dagligdag. Vi kender energiformen under betegnelsen ``Beliggenhedsenergi'' altså en energi som opstår som følge af et objekts beliggenhed.
	\end{block}\vfill
\end{frame}

\section{Potentiel energi i tyngdefeltet}
\begin{frame}{Potentiel energi}
En genstand som er hævet over jorden siges at have en potentiel energi i sammenligning med en genstand som ligger på jordens overflade.\vfill

For at gøre det nemt for os selv regner vi højder i forhold til havets overflade, derfor vil genstande som befinder sig ved havets overflade have en potentiel energi på 0 \joule.\vfill

\begin{theorem}
En genstand med massen $m$, som befinder sig i højden $h$ over havets overflade har den potentielle energi:
\begin{equation}
\notag E_\mathrm{pot} = m\cdot g\cdot h
\end{equation}
hvor $g$ er tyngdeaccelerationen som har værdien 9,82 \meter\per\second\squared.
\end{theorem}
\end{frame}

\begin{frame}{Potentiel energi}
\begin{block}{Et eksempel}
	Et æble med \alert{massen 100 \gram} hænger på en gren, i æbletræet \alert{7 \meter ~over jorden}. Bestem æblets potentielle energi.
\end{block}\pause\vfill
\begin{block}{Løsning}
	\begin{itemize}[<+->]
		\item Massen $m = 100 ~\gram = 0,100 ~\kilo\gram$
		\item Tyngdeaccelerationen $g = 9,82 ~\meter\per\second\squared$
		\item Højden $h = 7 ~\meter$
	\end{itemize}\pause
	\begin{align*}
		E_\mathrm{pot} &= m\cdot g\cdot h\\
			 &= 0,100 ~\kilo\gram\cdot 9,82 ~\meter\per\second\squared\cdot 7~\meter\\ 
			 &= 8,874 ~\joule
	\end{align*}
\end{block}
\end{frame}

\begin{frame}{Potentiel energi}
\begin{block}{Et eksempel}
	En potteplante med massen 2,2 \kilo\gram, som står i vinduet 90 \centi\meter ~over gulvet i en lejlighed, hvor gulvet ligger 12,0 \meter ~over gaden nedenfor.
\end{block}\pause\vfill
\begin{block}{Løsning}
	\begin{itemize}[<+->]
		\item Massen $m = 2,2 ~\kilo\gram$
		\item Tyngdeaccelerationen $g = 9,82 ~\meter\per\second\squared$
		\item Højden $h = (12,0 + 0,90) ~\meter$
	\end{itemize}\pause
	\begin{align*}
		E_\mathrm{pot} &= m\cdot g\cdot h\\
			 &= 2,2 ~\kilo\gram\cdot 9,82 ~\meter\per\second\squared\cdot 12,9~\meter\\ 
			 &= 280 ~\joule \approx 2,8\cdot 10^2 ~\joule
	\end{align*}
\end{block}
\end{frame}

\section{Kinetisk energi}
\begin{frame}{Kinetisk energi}
En genstand i bevægelse siges at have en kinetisk energi, i sammenligning med en genstand som er i hvile.\vfill

\begin{theorem}
En genstand med massen $m$ og som bevæger sig med farten $v$ har den kinetiske energi:\\
\begin{equation}
\notag E_\mathrm{kin} = \frac{1}{2}\cdot m\cdot v^2
\end{equation}
\end{theorem}\vfill
\end{frame}
